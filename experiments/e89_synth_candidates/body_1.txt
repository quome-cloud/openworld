# C9: Remove the vertical 3-stripe column
```python
def transform(grid):
    out=[row[:] for row in grid]
    cols={}
    for r in range(len(grid)):
        for c in range(len(grid[r])):
            if grid[r][c]==3: cols[c]=cols.get(c,0)+1
    stripe=set(cols)
    for r in range(len(out)):
        for c in range(len(out[r])):
            if c in stripe and out[r][c]==3: out[r][c]=0
    return out
```

```python
        for c in range(C):
```

# C21: Remove minority color (assume 3)
```python
def transform(grid):
    from collections import Counter
    cnt=Counter(v for row in grid for v in row if v!=0)
    target=3 if 3 in cnt else (min(cnt,key=cnt.get) if cnt else None)
    return [[0 if v==target else v for v in row] for row in grid]
```
