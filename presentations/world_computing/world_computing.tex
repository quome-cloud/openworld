\begin{frame}{Two kinds of world models}\vfill
\begin{columns}[T]
\begin{column}{0.46\textwidth}{\color{owmuted}\bfseries\large The picture-guessing kind}\par\medskip\begin{itemize}\small \item Guesses pixels (Dreamer, Sora)
\item Needs mountains of data
\item Small mistakes pile up
\item You can't read the insides\end{itemize}\end{column}
\begin{column}{0.46\textwidth}{\color{owteal}\bfseries\large The plain-code kind}\par\medskip\begin{itemize}\small \item A readable program over facts
\item The model writes it, then we check it
\item Right at every step, no matter how far
\item You can read the source\end{itemize}\end{column}
\end{columns}\vfill\end{frame}
\begin{frame}{Why we want checked code}\vfill\begin{enumerate}\item \textbf{Not a new rule} --- the model often already knows the rule; the win is elsewhere\item \textbf{Exact} --- no piling-up mistakes; a trained rule rots as (1-e)\textasciicircum{}t over time\item \textbf{Fast tool} --- runs way faster than asking the model each step; the agent just calls it\item \textbf{Like PyTorch} --- it didn't invent new models -- it made them quick to try out\end{enumerate}\vfill\end{frame}

\begin{frame}{What we add}\vfill\begin{itemize}\item \textbf{The idea} --- world computing, plus a way to snap worlds together that keeps the checking\item \textbf{OpenWorld} --- spell out the world, check the code, turn goals up or down with dials\item \textbf{Building} --- 100 worlds, 6 fields, minutes each, on an ordinary laptop\item \textbf{Share \& run} --- full-speed exact readable runs; trust checks and small shareable files\end{itemize}\vfill\end{frame}

\begin{frame}{Build big worlds from small checked pieces}\vfill\begin{columns}[c]\begin{column}{.42\textwidth}{\color{owdeep}\large\bfseries Small checked pieces}\\[3pt]{\color{owmuted}worlds are values; the connectors run each piece untouched -- bridges link, roll-ups summarize, injections pass values in}\end{column}\begin{column}{.10\textwidth}\begin{center}{\color{owochre}\Huge$\rightarrow$}\end{center}\end{column}\begin{column}{.42\textwidth}{\color{owdeep}\large\bfseries One checked whole}\\[3pt]{\color{owmuted}clean algebra (same in, same out, grouping doesn't matter); checked pieces make a checked whole -- one big n-rule job becomes n small ones}\end{column}\end{columns}\vfill\end{frame}
\begin{frame}{Writing the reward as checked code too}\vfill\begin{enumerate}\item \textbf{Reward as code} --- make the goal match the rules: a reward you can read and check\item \textbf{Same safe box} --- the reward code runs in the same little Python sandbox as the rules\item \textbf{Accept on match} --- only when it exactly matches rewards we held back to test with\item \textbf{Whole loop checked} --- seeing, rules, reward, planning -- what an ARC-AGI-3 'figure out the goal' puzzle needs\end{enumerate}\vfill\end{frame}

\begin{frame}{Closing the 'wrongly accepted' gap (E62)}\vfill\begin{enumerate}\item \textbf{The blind spot} --- a single quick test misses states you only reach later\item \textbf{Wrongly passed} --- on 3 faults that look clean at first, it wrongly passes them 100\% of the time\item \textbf{Full coverage fixes it} --- re-running across states that hit every branch drops wrong passes to 0\%\item \textbf{Across families} --- E16: Qwen, Llama, Gemma all reach a checked pass; Gemma-2-9B perfect every try, seconds per world\end{enumerate}\vfill\end{frame}

\begin{frame}{The big trade, plainly}\vfill
\begin{columns}[T]
\begin{column}{0.46\textwidth}{\color{owmuted}\bfseries\large Trained models}\par\medskip\begin{itemize}\small \item Cover lots of pixels
\item Pay in piling-up mistakes
\item Are unreadable\end{itemize}\end{column}
\begin{column}{0.46\textwidth}{\color{owteal}\bfseries\large Checked code}\par\medskip\begin{itemize}\small \item Exact, fast, works in new situations, inspectable
\item In plain-facts worlds, results all point one way
\item State it, check with invariants, probe intent, tighten, rebuild
\item Stays a program you can diff at every stage\end{itemize}\end{column}
\end{columns}\vfill\end{frame}
\begin{frame}{What these worlds unlock: world-time compute}\vfill\begin{itemize}\item \textbf{Free labels} --- checked worlds label training runs exactly, for free\item \textbf{Transfer} --- training on many worlds of a field lifts skill on worlds it never saw\item \textbf{World-time} --- the topic of the companion paper; exact, snappable worlds make it possible\item \textbf{Either way} --- but a world computer is useful even if you never train on it\end{itemize}\vfill\end{frame}
