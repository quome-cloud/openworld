\begin{frame}{The building block of computing with worlds}\centering\includegraphics[width=\linewidth,height=0.74\textheight,keepaspectratio]{prototyping_pipeline.png}\end{frame}
\begin{frame}{Two kinds of these world programs}
\begin{itemize}
  \item Picture-guessers: big neural nets that dream up pixels (Dreamer, Genie, Sora)
  \item Strong, but they need tons of data, their small mistakes pile up, and you cannot see inside
  \item Code worlds: the rule for what happens next is plain code you can read
  \item An AI writes the code, we check it is right before using it, and it is exact and easy to inspect
  \item The two kinds help each other; they are not rivals
\end{itemize}
\end{frame}
\begin{frame}{Why use code we check is right}
\begin{itemize}
  \item It runs exactly, fast, and you can audit it, and an agent can call it like a tool
  \item Same input, same answer: no piled-up mistakes; guessed rules rot as (1-e)\textasciicircum{}t
  \item Plain code plays out a world about 47,000x faster than asking an LLM each step
  \item At every step the thing you hold is code you can read and diff
\end{itemize}
\end{frame}

\begin{frame}{The plan, write, check relay}
\begin{itemize}
  \item You describe a world; a writer produces the rule: next = f(now, action)
  \item Check 1: it parses; Check 2: it runs safely and keeps the promises; Check 3: a critic reads it
  \item Accepted code is plain, editable code -- failures feed back into the next try
  \item Joining keeps the checks: checked pieces make a checked whole
\end{itemize}
\end{frame}

\begin{frame}{An agent that writes its own world program without reading the game's code}
\centering
\includegraphics[width=\linewidth,height=0.48\textheight,keepaspectratio]{arc3_architecture.png}\vskip3pt{\color{owmuted}\footnotesize See -> explore -> write code we check is right -> figure out the goal -> plan -> replay to verify\par}
\flushleft \vskip3pt{\footnotesize \begin{itemize}
  \item Never reads the game's code; learns the rules by playing
  \item Every win becomes a shareable OpenWorld world
  \item Copying the world is exact -- the hard part is figuring out the goal
\end{itemize}}
\end{frame}
\begin{frame}{The hard part: the win is a recipe of steps, not one screen}
\centering
\includegraphics[width=\linewidth,height=0.48\textheight,keepaspectratio]{arc3_goal_as_procedure.png}\vskip3pt{\color{owmuted}\footnotesize A win is an ordered recipe A->B->C, which no single-screen score can judge\par}
\flushleft \vskip3pt{\footnotesize \begin{itemize}
  \item Four careful ways to guess the goal solve 0 -- even with a perfect world copy
  \item Scoring one screen or blind searching cannot rank an ordered recipe
  \item Only an agent that reasons out the win gets past the walls
\end{itemize}}
\end{frame}
